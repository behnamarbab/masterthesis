    $trim(T)$\;

    \If{$deterministic\_mode$}{
      $deterministic\_stages(T)$\;
    }
    $MX \leftarrow calc\_stages(perf\_score, \mathcal{C})$\;
    \ForEach{element in $ (0, MX)$}{
      $random\_havoc(T)$\;
    }
    \caption{$fuzz\_one$: Fuzz one Entry}
    \label{algo:fuzzone}
\end{algorithm}
